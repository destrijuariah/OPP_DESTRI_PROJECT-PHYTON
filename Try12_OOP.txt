nilai_ujian = float(input("Masukan nilai ujian siswa: "))

if nilai_ujian >= 90:
  predikat = "Sangat Baik (A)"
elif nilai_ujian >= 75: 
  predikat = "Baik (B)"
elif nilai_ujian >= 60:
  predikat = "Cukup (C)"
else: 
  predikat = "Kurang (D) - Memerlukan Remedial"
  
print("Hasil Evaluasi: ", predikat)
